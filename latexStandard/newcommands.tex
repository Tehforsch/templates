%%% Brackets etc.
\newcommand{\bb}[1]{\left(#1\right)}                                        % (..)
\newcommand{\bc}[1]{\left[#1\right]}                                        % [..]
\newcommand{\bd}[1]{\left\langle#1\right\rangle}                            % <..>
\newcommand{\abs}[1]{\left|#1\right|}                                       % |..|
\newcommand{\abssq}[1]{\abs{#1}^2}                                          % |..|^2
\newcommand{\comm}[2]{\left[#1, #2\right]}                                  % [.., ..]
\newcommand{\anticomm}[2]{\left\{#1, #2\right\}}                            % {.., ..}
\newcommand{\bra}[1]{\ensuremath{\left\langle#1\right|}\xspace}             % <...|
\newcommand{\ket}[1]{\ensuremath{\left|#1\right\rangle}\xspace}             % |...>
\newcommand{\braket}[2]{\left\langle#1\middle|#2\right\rangle}              % <...|...>
\newcommand{\ketbra}[2]{\left|#1\middle\rangle\middle\langle#2\right|}      % |...><...|
\newcommand{\braketop}[3]{\left\langle#1\middle|#2\middle|#3\right\rangle}  % <...|...|...>

%%% Integrals, Differentials
\newcommand{\diff}[1]{\ensuremath{\frac{\mathrm{d}}{\mathrm{d}#1}}}         % d/dx
\newcommand{\sdiff}[1]{\ensuremath{\frac{\mathrm{d}^2}{\mathrm{d}#1^2}}}    % d^2/dx^2
\newcommand{\dd}[2]{\ensuremath{\frac{\mathrm{d}#1}{\mathrm{d}#2}}}         % df/dx
\newcommand{\sdd}[2]{\ensuremath{\frac{\mathrm{d}^2#1}{\mathrm{d}#2^2}}}    % d^2f/dx^2
\newcommand{\pdiff}[1]{\ensuremath{\frac{\partial}{\partial#1}}}            % partial d/dx
\newcommand{\spdiff}[1]{\ensuremath{\frac{\partial^2}{\partial#1^2}}}       % partial d^2/dx^2
\newcommand{\pp}[2]{\ensuremath{\frac{\partial#1}{\partial#2}}}             % partial df/dx
\newcommand{\spp}[2]{\ensuremath{\frac{\partial^2#1}{\partial#2^2}}}        % partial d^2f/dx^2

\newcommand{\integral}[4]{\int_{#1}^{#2} #4 \mathrm{d}#3}                   % Integral with non-italic differential
\newcommand{\infintegral}[2]{\int_{-\infty}^{\infty} #2 \mathrm{d}#1}       % Integral over entire R

%%% Constants
\newcommand{\kB}[0]{k_{\mathrm{B}}}                                         % Boltzmann factor
